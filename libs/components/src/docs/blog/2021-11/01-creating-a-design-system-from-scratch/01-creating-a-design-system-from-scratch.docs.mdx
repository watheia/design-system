### Component usage

```js
import Document from "./index"
const View = () => <Document />
```

### Render with theme and MDX providers

```js
import { ThemeCompositions } from "@watheia/docs.theme.theme-compositions"
import { MDXLayout } from "@watheia/docs.markdown.mdx"
import Document from "./index"
;<ThemeCompositions>
  <MDXLayout>
    <Document />
  </MDXLayout>
</ThemeCompositions>
```
